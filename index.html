// ========== 新增：手枪模型加载 ========== // 创建一个用于存放手枪的组，方便控制 const gunGroup = new THREE.Group(); // 手枪模型变量 let pistolModel = null; // 使用 Three.js 内置的 GLTFLoader 加载手枪模型 const loader = new THREE.GLTFLoader(); loader.load('lxxx.glb', function(gltf) { // 关键修改：不直接使用 gltf.scene，而是遍历它找到真正的手枪网格 let actualPistolMesh = null; // 遍历场景中的所有子对象 gltf.scene.traverse(function(child) { // 找到第一个 Mesh 对象，通常这就是我们要的手枪 if (child.isMesh) { actualPistolMesh = child; // 找到后就停止遍历，避免找到多个 return; } }); if (actualPistolMesh) { pistolModel = actualPistolMesh; // ========== 关键修复1：将模型的位置重置到原点 ========== pistolModel.position.set(0, 0, 0); // ========== 关键修复2：将模型的旋转重置为0 ========== pistolModel.rotation.set(0, 0, 0); // ========== 关键修复3：使用一个巨大的缩放值确保可见 ========== pistolModel.scale.set(1, 1, 1); // 先用1:1的缩放 // 将手枪添加到枪组中 gunGroup.add(pistolModel); // 将枪组添加到相机上 camera.add(gunGroup); // ========== 关键修复4：确保模型在相机前方 (Z为负值) ========== gunGroup.position.set(0, -1, -2); // Y轴向下，Z轴为负（前方） gunGroup.rotation.set(0, 0, 0); // 先不旋转枪组，看模型默认朝向 console.log("Actual Pistol Mesh found, reset position & rotation, and added:", pistolModel); } else { console.error("No Mesh found in the loaded GLTF model!"); } }, undefined, function(error) { console.error('An error occurred loading the pistol model:', error); }); // ========== 新增结束 ========
